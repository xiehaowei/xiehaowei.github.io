{% highlight shell %}
brew install lsof
#安装lsof
{% endhighlight %}

{% highlight shell %}
sudo lsof -nP -iTCP:端口号 -sTCP:LISTEN
#查看指定监听端口
{% endhighlight %}


{% highlight shell %}
sudo lsof -nP -iTCP -sTCP:LISTEN
# 查看端口监听情况 类似 netstat -tlnp
{% endhighlight %}

1. -n 表示不显示主机名
2. -P 表示不显示端口名称
